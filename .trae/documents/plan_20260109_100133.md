## 实现细节

### 1. 布局系统演示代码

```vue
<smart-form
  :model="layoutFormData"
  :fields="layoutFields"
  :itemSpan="12"
  label-width="100px"
>
  <!-- 表单内容 -->
</smart-form>

const layoutFields = [
  { name: 'field1', label: '字段1', type: 'input', span: 8 },
  { name: 'field2', label: '字段2', type: 'input', span: 16 },
  { name: 'field3', label: '字段3', type: 'input' }, // 使用默认 itemSpan: 12
  { name: 'field4', label: '字段4', type: 'input', span: 24 } // 占据整行
]
```

### 2. API 调用演示代码

```vue
<smart-form
  ref="formRef"
  :model="apiFormData"
  :fields="apiFields"
  label-width="100px"
>
  <div class="form-actions">
    <button @click="validateForm">验证表单</button>
    <button @click="resetForm">重置表单</button>
  </div>
</smart-form>

const formRef = ref()
const validateForm = async () => {
  const isValid = await formRef.value.validateForm()
  if (isValid) {
    alert('表单验证通过！')
  } else {
    alert('表单验证失败！')
  }
}
```

### 3. 适配器切换演示代码

```vue
<smart-form
  :model="adapterFormData"
  :fields="adapterFields"
  :adapter="currentAdapter"
  label-width="100px"
>
  <!-- 表单内容 -->
</smart-form>

<button @click="toggleAdapter">切换适配器</button>

const currentAdapter = ref('element')
const toggleAdapter = () => {
  currentAdapter.value = currentAdapter.value === 'element' ? 'ant' : 'element'
}
```
